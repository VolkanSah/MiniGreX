CREATE TABLE mgx_user (
  id INTEGER PRIMARY KEY AUTO_INCREMENT,
  username VARCHAR(255) NOT NULL,
  password VARCHAR(255) NOT NULL
);

CREATE TABLE mgx_post (
  id INTEGER PRIMARY KEY AUTO_INCREMENT,
  mgx_user_id INTEGER NOT NULL,
  link VARCHAR(255) NOT NULL,
  FOREIGN KEY (mgx_user_id) REFERENCES mgx_user(id)
);

CREATE TABLE comments (
  id INTEGER PRIMARY KEY AUTO_INCREMENT,
  mgx_post_id INTEGER NOT NULL,
  text VARCHAR(255) NOT NULL,
  FOREIGN KEY (mgx_post_id) REFERENCES mgx_post(id)
);

CREATE TABLE mgx_options (
  id INTEGER PRIMARY KEY AUTO_INCREMENT,
  mgx_user_id INTEGER NOT NULL,
  link VARCHAR(255) NOT NULL,
  FOREIGN KEY (mgx_user_id) REFERENCES mgx_user(id)
);
